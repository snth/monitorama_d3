draw some circles:


d3.select('svg')
    .selectAll('circle')
    .data(data)
    .enter()
    .append('circle')
      .attr('cx',function(d){return 180 + d.ll[1]})
      .attr('cy',function(d){return 90 + d.ll[0]})
      .attr('r',4)

using a key to join data:


ws.onmessage = function(evt){
    data = JSON.parse(evt.data); 
    circles = append_circles(data); 
    draw(circles);
}

function append_circles(data){
    var circles = d3.select('svg')
        .selectAll('circle')
        .data(data, function(d){return d.u + d.a})
        .enter()
        .append('circle');
    return circles;
}

draw a map of the world:

var projection = d3.geo.kavrayskiy7();

var path = d3.geo.path().projection(projection);

d3.json('/static/world-110m.json', function(d){
  d3.select('svg').selectAll("path")
      .data(topojson.object(d, d.objects.countries).geometries)
      .enter()
      .append('path')
        .attr('d', path)
        .style('fill','none')
        .style('stroke', 'gray')
})

and put it all together:

d3.select('svg').selectAll('circle')
    .data(data, function(d){return d.u+d.a})
    .enter()
    .append('circle')
    .attr('cx', function(d){
        projected = projection([d.ll[1], d.ll[0]])
        return projected[0]
    })
    .attr('cy', function(d){
        projected = projection([d.ll[1], d.ll[0]])
        return projected[1]
    })
    .attr('r', 4)
